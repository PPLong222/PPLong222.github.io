冒泡排序

void BubbleSort(int a[],int n) {
	for (int i = 0;i < n;i++) { 
		//设置了中断位，判断是否已经是排好序的数组了
		//能够实现O（n）的最优时间复杂度
		int isSwap = 0;
		for (int j = 0;j < n - i-1;j++) {
			if (a[j] > a[j+1]) {
				int temp = a[j];
				a[j] = a[j+1];
				a[j+1] = temp;
				isSwap = 1;
			}
		}
		
		for (int i = 0;i < n;i++) {
			printf(" %d ", a[i]);
		}
		printf("\n");
		if (isSwap == 0)
			return;
	}
	
}

选择排序

void SelectSort(int a[], int n) {
	int min; 
	int k=0;
	for (int i = 0;i < n;i++) {
		//最小的排在前面
		min = a[i];
		for (int j = i;j < n;j++) {
			if (a[j] < min) {
				min = a[j];
				k = j;
			}
		}
		if (min != a[i]) {
			a[k] = a[i];
			a[i] = min;
		}
	
		for (int i = 0;i < n;i++)
			printf(" %d ", a[i]);
		printf("\n");

	}

}

void SortByShell(int* arr, int len) {
	
	int i, j, increment;
	int temp;
	for (increment = len / 2;increment > 0;increment /= 2) {
		for (i = increment;i < len;i++) {
			temp = arr[i];
			for (j = i;j >= increment;j -= increment) {
				if (temp < arr[j - increment])
					arr[j] = arr[j - increment];
				else
					break;
			}
			//9,8,7,6,5,4,3,2,1,10 
			arr[j] = temp;
		}
	}
	printArr(arr, len);
}

void InsertSort(int a[], int n) {

	for (int i = 1;i < n;i++) {
		//插入排序前提是之前的数组都是排好序的了
		int j = i - 1;

//桶式排序
void BucketSort(int arr[], int num) {
	int MAX = findMax(arr, num)+1;
	// 为bucket分配空间，max尽可能的减少多余的空间
	int *bucket=(int*)calloc(MAX,sizeof(int));
	//为每一个arr的数设置标记为 1
	for (int i = 0;i < num;i++) {
		bucket[arr[i]]=1;
	}
	//打印
	for (int i = 0;i < MAX;i++) {
		if (bucket[i] == 1)
			printf("%d  ", i);
	}
}

// 基数排序
void RadixSort(int arr[], int num,int base) {
	int *count;
	//开辟有基数base个元素的数组
	count = (int*)malloc(sizeof(int)*base);
	// 开辟 base行 num列的数组
	int **temp;
	int maxCount = findMaxCount(arr, num);
	int countNum = 1;
	temp = (int**)malloc(sizeof(int*) * base);
	for (int i = 0;i < base;i++)
		temp[i] = (int*)malloc(num*sizeof(int));


	for (int a = 0;a < maxCount;a++) {
		
		//对count 和 temp进行 初始化
		for (int j = 0;j < base;j++) {
			count[j] = 0;
			for (int k = 0;k < num;k++)
					temp[j][k] = 0;
		}

		//对arr数组中每一个数 该位数上进行判断入桶
		for (int i = 0;i < num;i++) {
			int cNum = arr[i] / countNum;
			int leaveNum = cNum % base;
			// 进行标记
			count[leaveNum]++;
			//保存
			temp[leaveNum][count[leaveNum] - 1]=arr[i];
		}
		int j = 0;
		for (int i = 0;i < base;i++) {
			//寻找该位上是否有数（有标记）
			if (count[i] != 0) {
				for (int k = 0;k < count[i];k++) {
					arr[j] = temp[i][k];
					j++;
				}
			}
		}
		//每一个位数操作后对countNum*10 以便得到下一个位数
		countNum*= base;
	}
	/*for (int i = 0;i < num;i++)
		printf("%d ", arr[i]);
		*/
}

快速排序

public void quickSort(int[] nums,int l ,int r){
        if(l < r) {
        int index = getIndex(nums,l,r);
        quickSort(nums,l,index-1);
        quickSort(nums,index+1,r);
        }
    }
    public int getIndex(int[] nums,int l,int r){
        int temp = nums[l];
        int i = l;
        int j = r;
        while(i<j) {
            while (i < j && temp <= nums[j]) {
                j--;
            }
            nums[i] = nums[j];
            while (i < j && temp > nums[i]) {
                i++;
            }
            nums[j] = nums[i];
        }
        nums[i] = temp;
        return i ;
    }
