Response /ServletContext/ FileInputStream实现文件下载

  //获取 html参数名
String fileName=req.getParameter("filename");
//得到ServletContext对象
ServletContext servletContext=this.getServletContext();
//得到服务器包下的资源的绝对路径
String realPath=servletContext.getRealPath("/Sources/"+fileName);
//创建输入流实体
FileInputStream fileInputStream=new FileInputStream(realPath);
//更改响应头信息 设置为手动选择下载位置
resp.setHeader("content-type",servletContext.getMimeType(fileName));
resp.setHeader("content-disposition","attachment;filename="+fileName);
//得到输出流
ServletOutputStream outputStream = resp.getOutputStream();

byte[] readbyte=new byte[1024];
int len=0;
while((len=fileInputStream.read(readbyte))!=-1){
    outputStream.write(readbyte,0,len);
}
fileInputStream.close();
